\documentclass[]{standalone}

\usepackage{rotfig}

\begin{document}


\def\axisr{.05}
\begin{tikzpicture}[scale=3,font=\Large,
  aa/.style={
      axis,
      primarycolor,
      very thick,
      ->
    },
    ang/.style={
      angle,Accentcolor,thick
    }
  ]
  \draw[->,ang] (\axisr,-\axisr,.250) arc  (0:300:2*\axisr) ;

  \draw[aa] (0,0,0) -- (1,0,0) node[anchor=north east]{$\unitvec{x}$};
  \draw[aa] (0,0,0) -- (0,1,0) node[anchor=north west]{$\unitvec{y}$};
  \draw[aa] (0,0,0) -- (0,0,1)
  node[anchor=south]{$\unitvec{z}$};

  \draw[ang] (\axisr,-\axisr,.250) arc  (0:90:2*\axisr) ;


\end{tikzpicture}




\end{document}
